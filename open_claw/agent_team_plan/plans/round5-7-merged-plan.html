Ví dụ luồng CLI
# Human gửi yêu cầu cho Leader openclaw agent --agent agent_leader -m "TICKET: feature=main-gameplay, spec=concepts/main-gameplay.md, priority=High" # Leader tạo ticket + routing → dispatch GD openclaw agent --agent agent_gd -m "TASK_READY: ticket_id=ticket-20260320-001, sub_ticket=sub-20260320-001-01.md, feature=main-gameplay" # GD xong → báo Leader openclaw agent --agent agent_leader -m "STATUS_UPDATE: ticket_id=ticket-20260320-001, agent=agent_gd, status=done, output=design/GDD-FEATURE-main-gameplay.md" # Leader dispatch SONG SONG: QC1 + Dev Lead openclaw agent --agent agent_qc1 -m "TASK_READY: ticket_id=ticket-20260320-001, sub_ticket=sub-20260320-001-02.md, gdd_path=design/GDD-FEATURE-main-gameplay.md" openclaw agent --agent agent_dev -m "TASK_READY: ticket_id=ticket-20260320-001, sub_ticket=sub-20260320-001-03.md, gdd_path=design/GDD-FEATURE-main-gameplay.md" # Dev Lead dispatch SONG SONG: Server + Client openclaw agent --agent agent_dev_server -m "TASK_READY: ticket_id=ticket-20260320-001, sub_ticket=sub-20260320-001-04.md, feature=main-gameplay" openclaw agent --agent agent_dev_client -m "TASK_READY: ticket_id=ticket-20260320-001, sub_ticket=sub-20260320-001-05.md, feature=main-gameplay" # All done → Leader gọi QC2 → xong → Leader notify human # ✅ ticket-20260320-001 HOÀN THÀNH — GDD✅ Blackbox✅ Server✅ Client✅ Whitebox✅
